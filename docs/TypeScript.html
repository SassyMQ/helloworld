The TypeScript libraries have been tested with Angular2 and Ionic2.

Configuration

You'll need to add the the followin libraries, as well as adding references to each actor manually in the app.modules.ts.

  1. $ npm install @stomp/ng2-stompjs --save
  2. $ npm install angular2-uuid --save

Programmer


import { Component } from '@angular/core';
...
import { SMQProgrammer } from '../../../providers/sassymq/gss/smqProgrammer';
...
export class Home {

    constructor(public navCtrl: NavController, 
                public navParams: NavParams, 
                public programmer: SMQProgrammer) {

         // Try "Saying" some things
        
        
         // Say "Hello" to World
         // The Programmer, says "Hello" to the World
         var payload = this.programmer.createPayload();
         payload. = {...};
         this.programmer.Hello(payload, (reply) {
            // Do something here with reply.body.
         });
        
        
         // Say "Goodbye" to World
         // Say cya
         var payload = this.programmer.createPayload();
         payload. = {...};
         this.programmer.Goodbye(payload, (reply) {
            // Do something here with reply.body.
         });
        
        
         // Say "Get All Galaxies" to World
         // 
         var payload = this.programmer.createPayload();
         payload. = {...};
         this.programmer.GetAllGalaxies(payload, (reply) {
            // Do something here with reply.body.
         });
        

        
         // Listen for messages
        
        
         // Hear Whats Up from World
         // The World, says "Wassup?" back to the Programmer
         this.programmer.onWhatsUp((message) {
            // Do something here with message.body.
         });
        

    }

}

...
import { SMQWorld } from '../../../providers/sassymq/gss/smqWorld';
...
export class Home {

    constructor(public navCtrl: NavController, 
                public navParams: NavParams, 
                public world: SMQWorld) {

         // Try "Saying" some things
        
        
         // Say "Whats Up" to Programmer
         // The World, says "Wassup?" back to the Programmer
         var payload = this.world.createPayload();
         payload. = {...};
         this.world.WhatsUp(payload, (reply) {
            // Do something here with reply.body.
         });
        

        
         // Listen for messages
        
        
         // Hear Hello from Programmer
         // The Programmer, says "Hello" to the World
         this.world.onHello((message) {
            // Do something here with message.body.
         });
        
        
         // Hear Goodbye from Programmer
         // Say cya
         this.world.onGoodbye((message) {
            // Do something here with message.body.
         });
        
        
         // Hear Get All Galaxies from Programmer
         // 
         this.world.onGetAllGalaxies((message) {
            // Do something here with message.body.
         });
        
        
         // Hear Yo Yo from Universe
         // I'm ej
         this.world.onYoYo((message) {
            // Do something here with message.body.
         });
        

    }

}
